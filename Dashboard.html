Plot 1

plot_1 =
instacart_df %>% 
  group_by(aisle) %>%
  filter(n()> 10000) %>% 
  mutate(
    counts = n()
  ) %>% 
  ungroup() %>% 
  mutate(
    aisle = fct_reorder(aisle,counts)
  ) %>%
  ggplot(aes(y = aisle, fill = aisle)) + 
  geom_bar() +
  scale_x_continuous(trans = "sqrt")+
  theme(legend.position = "none") +
  labs(
    title = "Aisles with more than 10000 items ordered",
    x = "Counts(N)",
    y = "Aisle"
    )

ggplotly(plot_1)
